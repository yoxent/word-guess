    ```typescript
    export function getEndlessTotalWords(): number {
      return mmkv.getNumber(ENDLESS_TOTAL_KEY) ?? 0;
    }

    export function incrementEndlessTotalWords(): number {
      const current = getEndlessTotalWords();
      const next = current + 1;
      mmkv.set(ENDLESS_TOTAL_KEY, next);
      return next;
    }
    ```

    **Part B: Create leaderboardService.ts**

    Create `src/services/leaderboardService.ts` — coordinates score submission with offline/deferred queue support.

    **Exported functions:**

    ```typescript
    import firestoreService from './firestoreService';
    import * as syncQueue from './syncQueue';
    import { useAuthStore } from '../stores/authStore';
    import type { LeaderboardData } from '../types';

    type LeaderboardType = 'daily_streak' | 'endless_streak' | 'endless_total';

    /**
     * Submit a score to a specific leaderboard.
     * Tries Firestore directly if signed in. Falls back to sync queue if offline or not signed in.
     * The `forceQueue` parameter bypasses direct submission (used for deferred pre-auth scores).
     * Returns 'submitted' | 'queued' | 'failed'.
     */
    async function submitScore(
      type: LeaderboardType,
      score: number
    ): Promise<'submitted' | 'queued' | 'failed'>;

    /**
     * Called after a game completes. Determines which leaderboard(s) to update
     * based on game mode and outcome, then calls submitScore for each.
     * Uses the current auth state from authStore and game result data.
     *
     * For Daily Challenge win: submits daily streak (perModeStreaks.daily.current)
     * For Endless win: submits endless streak + endless total words
     * For Endless loss: submits final streak (may be 0) + endless total words
     */
    async function updateLeaderboardAfterGame(params: {
      mode: string;
      won: boolean;
      dailyStreak?: number;
      endlessStreak?: number;
      endlessTotalWords?: number;
    }): Promise<void>;

    /**
     * Fetch leaderboard data for a specific type.
     * Delegates to firestoreService.getLeaderboard() with error fallback.
     * Returns LeaderboardData with the signed-in player marked (isCurrentPlayer).
     */
    async function getLeaderboardData(type: LeaderboardType): Promise<LeaderboardData>;
    ```

    **Implementation details:**

    `submitScore`:
    1. Get auth state from `useAuthStore.getState()`
    2. If not signed in: `syncQueue.enqueueEvent('leaderboard_score', { type, score, ... })` and return 'queued'
    3. If signed in: try `firestoreService.submitLeaderboardScore(type, playerId, playerName, score)`
    4. If Firestore returns false (offline/error): `syncQueue.enqueueEvent(...)` and return 'queued'
    5. If Firestore returns true: return 'submitted'

    `updateLeaderboardAfterGame`:
    1. If mode is 'daily' and won: call `submitScore('daily_streak', params.dailyStreak ?? 0)`
    2. If mode is 'endless':
       - Call `submitScore('endless_streak', params.endlessStreak ?? 0)`
       - Call `submitScore('endless_total', params.endlessTotalWords ?? 0)`
    3. Fire-and-forget — no await in calling context (non-blocking per D-154)

    `getLeaderboardData`:
    1. Call `firestoreService.getLeaderboard(type)`
    2. If error or empty: return fallback LeaderboardData with empty entries
    3. Mark current player entries: compare `entry.playerId` with `useAuthStore.getState().playerId`
    4. Return LeaderboardData with updated entries

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| App ↔ Firestore (leaderboard writes) | Network — Firestore rules enforce auth-required writes |
| App ↔ Firestore (leaderboard reads) | Network — leaderboard reads are public (authenticated users only) |
| App ↔ Sync queue (AsyncStorage) | Local storage — app-private |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-ID-09 | Spoofing | Leaderboard score submission | Mitigated | Firestore rules: only authenticated users can write, user.uid must match document ID. Scores are computed client-side (accepted for MVP — see T-ID-10). |
| T-ID-10 | Tampering | Client-side score values | Accepted | Scores are computed on device. For MVP, trust client. Future: implement server-side validation with Cloud Functions. |
| T-ID-11 | Repudiation | Sync queue failure | Accepted | After 3 retries, events are discarded. Data loss accepted for extreme edge cases (D-148). Analytics and error reporting not implemented for MVP. |
| T-ID-12 | Information Disclosure | Player name in leaderboard | Accepted | Player display name from Google profile. Player chooses to sign in, opting into leaderboard visibility. No email or personal data exposed. |
</threat_model>
