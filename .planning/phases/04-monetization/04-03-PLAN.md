    ```typescript
    const showInterstitialIfNeeded = useCallback(async (): Promise<void> => {
      const { isPro: pro } = useSettingsStore.getState();
      if (pro) return; // Pro users skip interstitials (D-93, AD-05)
      
      const adStore = useAdStore.getState();
      const { gamesSinceLastAd } = adStore;
      
      // Frequency check (D-88):
      // Daily: every game (0 games since last ad = show now)
      // Endless/Random: every 2nd game (gamesSinceLastAd >= 1 means 2+ games since last ad)
      const mode = session?.mode;
      let threshold = 0;
      if (mode === 'daily') threshold = 0;
      else if (mode === 'endless' || mode === 'random') threshold = 1;
      
      if (gamesSinceLastAd >= threshold) {
        const shown = await adStore.showInterstitial();
        if (shown) {
          adStore.resetGamesSinceLastAd();
        }
      }
    }, [session?.mode]);
    ```
