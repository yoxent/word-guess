    function ToggleRow({ config }: { config: SettingsRowConfig & { type: 'toggle' } }) {
      const value = useSettingsStore((s) => s[config.storeKey]);
      const toggleAction = useSettingsStore((s) => {
        // Map config.storeKey to the corresponding toggle action
        switch (config.storeKey) {
          case 'hardModeEnabled': return s.toggleHardMode;
          case 'soundEnabled': return s.toggleSound;
          case 'hapticEnabled': return s.toggleHaptic;
          default: return () => {};
        }
      });

      return (
        <View style={styles.row}>
          <Text style={styles.label}>{config.label}</Text>
          <Switch
            value={!!value}
            onValueChange={toggleAction}
            trackColor={{ false: colors.tileEmpty, true: colors.accent }}
            thumbColor={colors.textInverse}
          />
        </View>
      );
    }
    ```

    export interface GameResultForShare {
      mode: GameMode;
      word: string;
      attempts: number;
      won: boolean;
      maxAttempts: number;
      guesses: GuessFeedback[][];   // each guess's per-tile feedback
      date: string;                  // ISO date string
    }
    ```

    **Emoji map:**
    ```typescript
    const TILE_EMOJI: Record<string, string> = {
      correct: '🟩',
      present: '🟨',
      absent: '⬛',
    };
    ```

    **Mode name display map:**
    ```typescript
    const MODE_NAMES: Record<GameMode, string> = {
      free: 'Free Play',
      random: 'Random',
      daily: 'Daily Challenge',
      endless: 'Endless',
    };
    ```

    **generateShareText function:**

    ```typescript
    export function generateShareText(result: GameResultForShare): string {
      const modeName = MODE_NAMES[result.mode] || 'Word Guess';
      const dateFormatted = result.date.split('T')[0]; // YYYY-MM-DD
      const lines: string[] = [];

      // Header
      lines.push(`Word Guess - ${modeName}`);
      lines.push(dateFormatted);
      lines.push('');

      // Emoji rows
      for (const guess of result.guesses) {
        const row = guess.map((f) => TILE_EMOJI[f.feedback] || '⬛').join('');
        lines.push(row);
      }

      // Attempt counter
      if (result.won) {
        lines.push(`${result.attempts}/${result.maxAttempts}`);
      } else {
        lines.push(`X/${result.maxAttempts}`);
      }

      // Footer
      lines.push('');
      lines.push('Play Word Guess!');

      return lines.join('\n');
    }
    ```

    **Edge cases handled:**
    - Empty guesses array: returns header + date + "X/{maxAttempts}" + footer
    - Win with 1 guess: single emoji row + "1/{maxAttempts}"
    - Loss: guess rows up to maxAttempts + "X/{maxAttempts}"
    - Unknown mode: falls back to "Word Guess" header
    - No feedback for a guess: all absent (⬛)

    **Format example (per UI-SPEC):**
    ```
    Word Guess - Daily Challenge
    2026-07-05

    🟩🟩🟩🟩🟩
    5/6

    Play Word Guess!
    ```

    **Loss format:**
    ```
    Word Guess - Random
    2026-07-05

    ⬛🟩⬛⬛🟨
    🟨🟨⬛🟩⬛
    X/6

    Play Word Guess!
    ```

    Utility is pure (no side effects, no React imports) — can be tested standalone.
