     }

     Animated.timing(titleAnim, {

     }).start();

     Animated.stagger(HOME_STAGGER_DELAY, buttonAnims.map(anim =>

         useNativeDriver: true, easing: Easing.out(Easing.ease),
       })
     )).start();

     // Icons start after stagger completes
     const lastButtonDelay = HOME_STAGGER_DELAY * 3 + HOME_STAGGER_DURATION;
     const iconTimer = setTimeout(() => {
       Animated.timing(iconAnim, {
         toValue: 1, duration: HOME_STAGGER_DURATION,
         useNativeDriver: true, easing: Easing.out(Easing.ease),

     return () => clearTimeout(iconTimer);
   }, [reduceMotion, titleAnim, buttonAnims, iconAnim]);
